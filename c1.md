raylib를 이용해서 5번 실패하면 실행종료가 됨

#include "raylib.h"
#include <stdbool.h>

#define SCREEN_WIDTH 800
#define SCREEN_HEIGHT 600

#define BRICK_ROWS 6
#define BRICK_COLS 10
#define BRICK_WIDTH 70
#define BRICK_HEIGHT 25

typedef struct {
    Rectangle rect;
    bool active;
    Color color;
} Brick;

int main(void)
{
    InitWindow(SCREEN_WIDTH, SCREEN_HEIGHT, "Rainbow Breakout");

    SetTargetFPS(60);

    // Paddle
    Rectangle paddle = {
        SCREEN_WIDTH/2 - 60,
        SCREEN_HEIGHT - 50,
        120,
        20
    };

    // Ball
    Vector2 ball = {
        SCREEN_WIDTH/2,
        SCREEN_HEIGHT/2
    };

    Vector2 ballSpeed = {5, -5};
    float ballRadius = 10;

    // Lives
    int lives = 5;

    // Bricks
    Brick bricks[BRICK_ROWS][BRICK_COLS];

    Color rainbow[6] = {
        RED,
        ORANGE,
        YELLOW,
        GREEN,
        BLUE,
        PURPLE
    };

    for (int row = 0; row < BRICK_ROWS; row++)
    {
        for (int col = 0; col < BRICK_COLS; col++)
        {
            bricks[row][col].rect = (Rectangle){
                40 + col * (BRICK_WIDTH + 5),
                50 + row * (BRICK_HEIGHT + 5),
                BRICK_WIDTH,
                BRICK_HEIGHT
            };

            bricks[row][col].active = true;
            bricks[row][col].color = rainbow[row % 6];
        }
    }

    bool gameOver = false;
    bool win = false;

    while (!WindowShouldClose())
    {
        if (!gameOver && !win)
        {
            // Paddle movement
            if (IsKeyDown(KEY_LEFT))
                paddle.x -= 8;

            if (IsKeyDown(KEY_RIGHT))
                paddle.x += 8;

            if (paddle.x < 0)
                paddle.x = 0;

            if (paddle.x + paddle.width > SCREEN_WIDTH)
                paddle.x = SCREEN_WIDTH - paddle.width;

            // Ball movement
            ball.x += ballSpeed.x;
            ball.y += ballSpeed.y;

            // Wall collisions
            if (ball.x <= ballRadius || ball.x >= SCREEN_WIDTH - ballRadius)
                ballSpeed.x *= -1;

            if (ball.y <= ballRadius)
                ballSpeed.y *= -1;

            // Paddle collision
            if (CheckCollisionCircleRec(ball, ballRadius, paddle))
            {
                ballSpeed.y = -fabsf(ballSpeed.y);

                float hitPos =
                    (ball.x - (paddle.x + paddle.width / 2))
                    / (paddle.width / 2);

                ballSpeed.x = hitPos * 7;
            }

            // Brick collision
            int remainingBricks = 0;

            for (int r = 0; r < BRICK_ROWS; r++)
            {
                for (int c = 0; c < BRICK_COLS; c++)
                {
                    if (bricks[r][c].active)
                    {
                        remainingBricks++;

                        if (CheckCollisionCircleRec(
                                ball,
                                ballRadius,
                                bricks[r][c].rect))
                        {
                            bricks[r][c].active = false;
                            ballSpeed.y *= -1;
                            remainingBricks--;
                        }
                    }
                }
            }

            // Win condition
            if (remainingBricks == 0)
            {
                win = true;
            }

            // Ball lost
            if (ball.y > SCREEN_HEIGHT)
            {
                lives--;

                if (lives <= 0)
                {
                    gameOver = true;
                }
                else
                {
                    ball = (Vector2){
                        SCREEN_WIDTH/2,
                        SCREEN_HEIGHT/2
                    };

                    ballSpeed = (Vector2){5, -5};

                    paddle.x = SCREEN_WIDTH/2 - 60;
                }
            }
        }

        BeginDrawing();

        ClearBackground(BLACK);

        // Draw bricks
        for (int r = 0; r < BRICK_ROWS; r++)
        {
            for (int c = 0; c < BRICK_COLS; c++)
            {
                if (bricks[r][c].active)
                {
                    DrawRectangleRec(
                        bricks[r][c].rect,
                        bricks[r][c].color
                    );
                }
            }
        }

        // Draw paddle
        DrawRectangleRec(paddle, WHITE);

        // Draw ball
        DrawCircleV(ball, ballRadius, SKYBLUE);

        // Lives display
        DrawText(TextFormat("Lives: %d", lives),
                 20, 20, 25, WHITE);

        if (gameOver)
        {
            DrawText("GAME OVER",
                     SCREEN_WIDTH/2 - 120,
                     SCREEN_HEIGHT/2 - 30,
                     50,
                     RED);

            DrawText("You used all 5 chances!",
                     SCREEN_WIDTH/2 - 140,
                     SCREEN_HEIGHT/2 + 30,
                     25,
                     WHITE);
        }

        if (win)
        {
            DrawText("YOU WIN!",
                     SCREEN_WIDTH/2 - 100,
                     SCREEN_HEIGHT/2,
                     50,
                     GREEN);
        }

        EndDrawing();
    }

    CloseWindow();

    return 0;
}
