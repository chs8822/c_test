구조체를 활용하여 2명의 이름과 성적을 초기화하고 과목평균과 학생평균을 화면에 출력하는 코드를 c언어로 작성하시오

#include <stdio.h>

#define SUBJECTS 3
#define STUDENTS 2

typedef struct {
    char name[20];
    int score[SUBJECTS];
    float avg;
} Student;

int main() {
    Student students[STUDENTS] = {
        {"Kim", {90, 85, 88}},
        {"Lee", {80, 95, 75}}
    };

    int i, j;

    // 학생별 평균 계산
    for (i = 0; i < STUDENTS; i++) {
        int sum = 0;
        for (j = 0; j < SUBJECTS; j++) {
            sum += students[i].score[j];
        }
        students[i].avg = (float)sum / SUBJECTS;
    }

    // 학생별 평균 출력
    printf("=== 학생별 평균 ===\n");
    for (i = 0; i < STUDENTS; i++) {
        printf("%s의 평균: %.2f\n",
               students[i].name,
               students[i].avg);
    }

    // 과목별 평균 계산 및 출력
    printf("\n=== 과목별 평균 ===\n");
    for (j = 0; j < SUBJECTS; j++) {
        int sum = 0;
        for (i = 0; i < STUDENTS; i++) {
            sum += students[i].score[j];
        }
        printf("과목 %d 평균: %.2f\n",
               j + 1,
               (float)sum / STUDENTS);
    }

    return 0;
}
